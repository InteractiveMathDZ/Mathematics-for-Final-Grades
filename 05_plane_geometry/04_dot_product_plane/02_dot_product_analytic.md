elements:
  - h3: "1. العبارة التحليلية للجداء السلمي"
    id: "analytic-dot-formula"

  - parags:
      - rtl: "في معلم متعامد ومتجانس $(O; \\vec{i}, \\vec{j})$، إذا كان لدينا شعاعان $\\vec{u}=\\binom{x}{y}$ و $\\vec{v}=\\binom{x'}{y'}$، فإن الجداء السلمي بينهما يحسب بالعلاقة التالية:"
      - center: "$$\\vec{u} \\cdot \\vec{v} = x \\cdot x' + y \\cdot y'$$"
      - rtl: "<strong>ملاحظة منهجية:</strong> هذه العبارة لا تنطبق إلا إذا كان المعلم متعامداً ومتجانساً."

  - h3: "2. حساب الطويلة والزوايا تحليلياً"
    id: "analytic-norm-distance"

  - parags:
      - rtl: "بالاعتماد على أن المربع السلمي للشعاع يساوي مربع طويلته ($\\vec{u}^2 = \\|\\vec{u}\\|^2$)، تسمح العبارة التحليلية بحساب الأطوال وقيس الزوايا كالتالي:"
      - rtl: "- <strong>طويلة الشعاع $\\vec{u}$:</strong> هي جذر مجموع مربعي مركبتيه: $\\|\\vec{u}\\| = \\sqrt{x^2 + y^2}$."
      - rtl: "- <strong>حساب جيب تمام زاوية ($\\cos \\theta$):</strong> بدمج العبارتين الهندسية والتحليلية نجد:"
      - center: "$$\\cos(\\theta) = \\frac{x \\cdot x' + y \\cdot y'}{\\sqrt{x^2 + y^2} \\times \\sqrt{x'^2 + y'^2}}$$"

  - h3: "3. الشرط التحليلي لتعامد شعاعين"
    id: "analytic-orthogonality"

  - tableLines:
      dir: rtl
      lines:
        - - th: "الوضعية الهندسية"
          - th: "الشرط التحليلي"
          - th: "النتيجة الهندسية"
        - - th: "$\\vec{u} \\perp \\vec{v}$"
          - td: "$x \\cdot x' + y \\cdot y' = 0$"
          - td: "يكون الشعاعان غير المعدومين متعامدين إذا وفقط إذا انعدم مجموع جداء مركباتهما المتناظرة."

  - h3: "4. مثال تطبيقي (حساب الجداء السلمي واستنتاج التعامد)"
    id: "dot-analyt-ex"

  - parags:
      - rtl: "في معلم متعامد ومتجانس، نعتبر الشعاعين $\\vec{u}=\\binom{2}{3}$ و $\\vec{v}=\\binom{-6}{4}$. احسب $\\vec{u} \\cdot \\vec{v}$، ثم استنتج طبيعة الزاوية المحصورة بينهما."

  - tableLines:
      dir: rtl
      lines:
        - - th: "المرحلة الحسابية"
          - th: "التطبيق والتعويض التحليلي"
          - th: "النتيجة"
        - - th: "1. حساب الجداء السلمي"
          - td: "$\\vec{u} \\cdot \\vec{v} = (2 \\times -6) + (3 \\times 4) = -12 + 12$"
          - td: "$\\vec{u} \\cdot \\vec{v} = 0$"
        - - th: "2. استنتاج الوضعية"
          - td: "بما أن الجداء السلمي معدوم، فإن الشعاعين متعامدان."
          - td: "$\\vec{u} \\perp \\vec{v}$"
        - - th: "3. قيس الزاوية $\\theta$"
          - td: "بما أن التعامد محقق، فإن جيب تمام الزاوية ينعدم ($\\cos \\theta = 0$)."
          - td: "$\\theta = 90^\\circ$ (زاوية قائمة)"
