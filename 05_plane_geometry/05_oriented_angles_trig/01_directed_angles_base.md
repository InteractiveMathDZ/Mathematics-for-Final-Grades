  - parags:

  - h3: "3. القياس الرئيسي لزاوية موجهة"
    id: "principal-value"

  - parags:
      - rtl: "من بين القياسات العديدة للزاوية الموجهة، يوجد قياس وحيد $\\alpha$ ينتمي إلى المجال نصف المفتوح $]-\\pi ; \\pi]$، ويسمى هذا القياس بـ <strong>القياس الرئيسي</strong> للزاوية الموجهة."

  - tableLines:
      dir: rtl
      lines:
        - - th: "قيمة القياس الرئيسي $\\alpha$"
          - th: "اتجاه الحركة"
          - th: "التفسير الهندسي"
        - - th: "$\\alpha > 0$"
          - td: "حركة في الاتجاه المباشر."
          - td: "الزاوية تتشكل عكس اتجاه عقارب الساعة."
        - - th: "$\\alpha < 0$"
          - td: "حركة في الاتجاه غير المباشر."
          - td: "الزاوية تتشكل في اتجاه عقارب الساعة."
        - - th: "$\\alpha = \\pi$"
          - td: "نصف دورة."
          - td: "الشعاعان متعاكسان في الاتجاه (زاوية مستقيمة)."

  - tableLines:
      dir: rtl
      lines:
        - - th: "الخطوة الحسابية"
          - th: "التطبيق العددي"
          - th: "التبرير الرياضي"
        - - th: "1. البحث عن المضاعف الزوجي الأقرب"
          - td: "$\\frac{17\\pi}{3} = \\frac{18\\pi - \\pi}{3}$"
          - td: "العدد $18$ هو أقرب مضاعف للمقام $3$ يعطي ناتجاً زوجياً $(18/3=6).$"
        - - th: "2. تفكيك الكسر واختزاله"
          - td: "$\\theta = \\frac{18\\pi}{3} - \\frac{\\pi}{3} = 6\\pi - \\frac{\\pi}{3}$"
          - td: "عزل عدد الدورات الكاملة: $6\\pi = 3 \\times (2\\pi)$ (أي $3$ دورات كاملة)."
        - - th: "3. التحقق من الانتماء للمجال"
          - td: "$\\alpha = -\\frac{\\pi}{3}$"
          - td: "بما أن $-\\frac{\\pi}{3} \\in ]-\\pi ; \\pi]$، فإن هذا هو القياس الرئيسي للزاوية."
