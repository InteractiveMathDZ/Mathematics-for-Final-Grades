elements:
  - h3: "1. علاقة شال للزوايا الموجهة"
    id: "chasles-relation-angles"

  - parags:
      - rtl: "تسمح علاقة شال للزوايا الموجهة بحساب قيس زاوية موجهة لشعاعين بالاعتماد على شعاع وسيط $\\vec{w}$ غير معدوم. يكون قيس الزاوية الإجمالية مساوياً لمجموع قيسي الزاويتين الناتجين (بترديد $2\\pi$):"
      - center: "$(\\vec{u}, \\vec{v}) = (\\vec{u}, \\vec{w}) + (\\vec{w}, \\vec{v}) + 2k\\pi \\quad \\text{حيث } k \\in \\mathbb{Z}$"
      - rtl: "<strong>ملاحظة منهجية:</strong> تُستخدم هذه الخاصية لتبسيط العبارات الزاوية وحساب أقيايس الزوايا المركبة في الأشكال الهندسية."

  - h3: "2. أثر تغيير ترتيب أو اتجاه الأشعة على قيس الزاوية"
    id: "sign-effects"

  - parags:
      - rtl: "تتغير قيم وإشارات أقيايس الزوايا الموجهة لشعاعين تبعاً لتغيير ترتيب الشعاعين أو اتجاههما وفق الخواص الحسابية التالية:"

  - tableLines:
      dir: rtl
      lines:
        - - th: "التحويل الشعاعي للزاوية"
          - th: "العلاقة الحسابية"
          - th: "التفسير الهندسي"
        - - th: "عكس ترتيب الشعاعين"
          - td: "$(\\vec{v}, \\vec{u}) = -(\\vec{u}, \\vec{v}) + 2k\\pi$"
          - td: "تغيير ترتيب الشعاعين يؤدي إلى تغيير إشارة قيس الزاوية الموجهة."
        - - th: "معاكس الشعاع الثاني"
          - td: "$(\\vec{u}, -\\vec{v}) = (\\vec{u}, \\vec{v}) + \\pi + 2k\\pi$"
          - td: "عكس اتجاه الشعاع الثاني يؤدي إلى إضافة القيمة $\\pi$ لقيس الزاوية."
        - - th: "معاكس الشعاع الأول"
          - td: "$(-\\vec{u}, \\vec{v}) = (\\vec{u}, \\vec{v}) + \\pi + 2k\\pi$"
          - td: "عكس اتجاه الشعاع الأول يؤدي إلى إضافة القيمة $\\pi$ لقيس الزاوية."
        - - th: "معاكس الشعاعين معاً"
          - td: "$(-\\vec{u}, -\\vec{v}) = (\\vec{u}, \\vec{v}) + 2k\\pi$"
          - td: "عكس اتجاه كلا الشعاعين يحافظ على قيس الزاوية الأصلي دون تغيير."

  - h3: "3. شرطا التوازي والتعامد بالزوايا الموجهة"
    id: "geometric-conditions-angles"

  - tableLines:
      dir: rtl
      lines:
        - - th: "الوضعية الهندسية"
          - th: "قيس الزاوية الموجهة $(\\vec{u}, \\vec{v})$"
          - th: "التفسير"
        - - th: "الارتباط الخطي (التوازي أو الاستقامية)"
          - td: "$(\\vec{u}, \\vec{v}) = 0 \\quad \\text{أو} \\quad (\\vec{u}, \\vec{v}) = \\pi$"
          - td: "يكون الشعاعان في نفس الاتجاه إذا كان القيس $0$، وفي اتجاهين متعاكسين إذا كان القيس $\\pi$."
        - - th: "التعامد"
          - td: "$(\\vec{u}, \\vec{v}) = \\frac{\\pi}{2} \\quad \\text{أو} \\quad (\\vec{u}, \\vec{v}) = -\\frac{\\pi}{2}$"
          - td: "يكون حاملا الشعاعين متعامدين (الزاوية قائمة في الاتجاه المباشر أو غير المباشر)."

  - h3: "4. مثال تطبيقي (تبسيط عبارة زاوية موجهة باستخدام الخواص)"
    id: "angles-prop-ex"

  - parags:
      - rtl: "علماً أن قيس الزاوية الموجهة $(\\vec{u}, \\vec{v}) = \\frac{\\pi}{4}$، احسب القياس الرئيسي للزاوية الموجهة التالية: $\\theta = (-\\vec{v}, \\vec{u})$."

  - tableLines:
      dir: rtl
      lines:
        - - th: "المرحلة"
          - th: "تطبيق الخواص الحسابية"
          - th: "التبرير الرياضي"
        - - th: "1. معالجة إشارة الشعاع الأول"
          - td: "$(-\\vec{v}, \\vec{u}) = (\\vec{v}, \\vec{u}) + \\pi$"
          - td: "عكس اتجاه الشعاع الأول يضيف القيمة $\\pi$ للقيس."
        - - th: "2. قلب ترتيب الشعاعين"
          - td: "$(\\vec{v}, \\vec{u}) = -(\\vec{u}, \\vec{v}) = -\\frac{\\pi}{4}$"
          - td: "تغيير ترتيب الشعاعين يعكس إشارة القيس المعطى."
        - - th: "3. الحساب العددي"
          - td: "$\\theta = -\\frac{\\pi}{4} + \\pi = -\\frac{\\pi}{4} + \\frac{4\\pi}{4} = \\frac{3\\pi}{4}$"
          - td: "توحيد المقامات وحساب القيمة النهائية."
        - - th: "4. التحقق من القياس الرئيسي"
          - td: "$\\frac{3\\pi}{4} \\in ]-\\pi ; \\pi]$"
          - td: "بما أن النتيجة تنتمي للمجال المعين، فإنها تمثل القياس الرئيسي للزاوية."
