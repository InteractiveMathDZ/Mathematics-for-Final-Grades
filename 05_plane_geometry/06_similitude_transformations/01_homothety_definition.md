elements:
  - h3: "1. التعريف الهندسي للتحاكي"
    id: "homothety-definition"

  - parags:
      - rtl: "التحاكي الذي مركزه النقطة المعلومة $\\Omega$ ونسبته العدد الحقيقي غير المعدوم $k$ (حيث $k \\neq 0$)، هو تحويل نقطي يرفق بكل نقطة $M$ من المستوي نقطة $M'$ تحقق العلاقة الشعاعية التالية:"
      - center: "$$\\vec{\\Omega M'} = k \\cdot \\vec{\\Omega M}$$"
      - rtl: "ونرمز له بالرمز $h(\\Omega, k)$، ونكتب: $h(M) = M'$."

  - h3: "2. المناقشة البيانية لنسبة التحاكي $k$"
    id: "homothety-ratio-analysis"

  - parags:
      - rtl: "تتحدد وضعية النقطة الصورة $M'$ بالنسبة إلى النقطة الأصلية $M"

      - rtl: " ومركزه $\\Omega$ تبعاً لقيمة النسبة $k$ وإشارتها كما يلي:"

  - tableLines:
      dir: rtl
      lines:
        - - th: "قيمة النسبة $k$"
          - th: "الوضعية الشعاعية والهندسية"
          - th: "طبيعة التحويل"
        - - th: "$k > 0$"
          - td: "الشعاعان $\\vec{\\Omega M'}$ و $\\vec{\\Omega M}$ في نفس الاتجاه. النقطة $\\Omega$ خارج القطعة $[MM']$."
          - td: "تمديد (إذا كان $k > 1$) أو تقليص (إذا كان $0 < k < 1$)."
        - - th: "$k < 0$"
          - td: "الشعاعان $\\vec{\\Omega M'}$ و $\\vec{\\Omega M}$ في اتجاهين متعاكسين. النقطة $\\Omega$ تنتمي للقطعة $[MM']$."
          - td: "تحويل يتضمن تكبيراً أو تصغيراً مع قلب اتجاه الشكل."
        - - th: "$k = 1$"
          - td: "$\\vec{\\Omega M'} = \\vec{\\Omega M} \\implies M' = M$."
          - td: "تحويل متطابق (كل نقطة هي صورة نفسها)."
        - - th: "$k = -1$"
          - td: "$\\vec{\\Omega M'} = -\\vec{\\Omega M}$، وتكون $\\Omega$ منتصف القطعة $[MM']$."
          - td: "تناظر مركزي مركزه النقطة $\\Omega$."

  - h3: "3. خوارزمية الإنشاء الهندسي لصورة نقطة"
    id: "homothety-construction"

  - parags:
      - rtl: "لإنشاء صورة النقطة $M$ بالتحاكي $h(\\Omega, k)$، نتبع الخطوات الهندسة الجبرية التالية:"

  - tableLines:
      dir: rtl
      lines:
        - - th: "الخطوة"
          - th: "طريقة العمل والإنشاء"
          - th: "التعليل الرياضي"
        - - th: "1. رسم حامل الشعاع"
          - td: "نرسم المستقيم $(\\Omega M)$ المار بالمركز والنقطة الأصلية."
          - td: "النقط $\\Omega$، $M$ و $M'$ في استقامية حتماً بناءً على الارتباط الخطي للأشعة."
        - - th: "2. تحديد جهة الحركة"
          - td: "إذا كان $k > 0$ نتحرك من $\\Omega$ في جهة $M$. وإذا كان $k < 0$ نتحرك في الجهة المعاكسة."
          - td: "إشارة العدد $k$ تحدد اتجاه الشعاعين المتناسبين."
        - - th: "3. تعيين المسافة الدقيقة"
          - td: "نحسب الطول $\\Omega M' = |k| \\times \\Omega M$ ونعلم النقطة $M'$ باستخدام الفرجار أو المسطرة المدرجة."
          - td: "طويلة الشعاع الناتج تساوي القيمة المطلقة للنسبة مضروبة في طويلة الشعاع الأصلي."

  - h3: "4. مثال تطبيقي (إنشاء صورة قطعة مستقيمة بالتحاكي)"
    id: "homothety-example"

  - parags:
      - rtl: "لتكن $[AB]$ قطعة مستقيمة طولها $4\\text{ cm}$، و $\\Omega$ نقطة خارج المستقيم $(AB)$. أنشئ القطعة $[A'B']$ صورة $[AB]$ بالتحاكي $h$ الذي مركزه $\\Omega$ ونسبته $k = -1.5$."

  - tableLines:
      dir: rtl
      lines:
        - - th: "المرحلة"
          - th: "خطوات الإنشاء والتعويض"
          - th: "النتيجة الهندسية"
        - - th: "1. إنشاء النقطة $A'$"
          - td: "نرسم المستقيم $(\\Omega A)$، وبما أن $k=-1.5$، نحدد $A'$ في الجهة المعاكسة لـ $A$ بالنسبة إلى $\\Omega$ بحيث: $\\Omega A' = 1.5 \\times \\Omega A$."
          - td: "$h(A) = A'$"
        - - th: "2. إنشاء النقطة $B'$"
          - td: "نرسم المستقيم $(\\Omega B)$، ونحدد $B'$ في الجهة المعاكسة لـ $B$ بالنسبة إلى $\\Omega$ بحيث: $\\Omega B' = 1.5 \\times \\Omega B$."
          - td: "$h(B) = B'$"
        - - th: "3. استنتاج طول الصورة"
          - td: "نصل بين النقطتين $A'$ و $B'$ لتتشكل القطعة $[A'B']$. يحسب طولها بالعلاقة: $A'B' = |-1.5| \\times AB = 1.5 \\times 4$."
          - td: "$A'B' = 6\\text{ cm}$ والقطعة $[A'B']$ توازي $[AB]$."

  - NB:
      type: success
      title: "قاعدة:"
      NBs:
        - rtl: "التعريف الشعاعي للتحاكي يفرض أن المركز والنقطة وصورتها على استقامية واحدة دائماً؛ والنسبة $k$ تتحكم في المسافة والاتجاه معاً."
