elements:
  - h3: "1. التعريف الهندسي للدوران"
    id: "rotation-definition"

  - parags:
      - rtl: "الدوران $r$ الذي مركزه النقطة المعلومة $\\Omega$ وزاويته الموجهة $\\theta$، هو تحويل نقطي يحول النقطة $\\Omega$ إلى نفسها، ويرفق بكل نقطة $M$ تختلف عن $\\Omega$ نقطة $M'$ حيث تحققان الشرطين التاليين معاً:"
      - center: "$\\Omega M' = \\Omega M \\quad \\text{و} \\quad (\\vec{\\Omega M}, \\vec{\\Omega M'}) = \\theta + 2k\\pi \\quad (k \\in \\mathbb{Z})$"

      - rtl: "ونرمز له بالرمز $r(\\Omega, \\theta)$، ونكتب: $r(M) = M'$."

  - h3: "2. حالات خاصة لزاوية الدوران $\\theta$"
    id: "rotation-special-cases"

  - parags:
      - rtl: "تؤول طبيعة الدوران إلى تحويلات نقطية مألوفة بناءً على قيم الزاوية $\\theta$ كالتالي:"

  - tableLines:
      dir: rtl
      lines:
        - - th: "قيمة الزاوية $\\theta$"
          - th: "الشرط الشعاعي والهندسي"
          - th: "طبيعة التحويل الناتجة"
        - - th: "$\\theta = 0$"
          - td: "$\\Omega M' = \\Omega M$ و $(\\vec{\\Omega M}, \\vec{\\Omega M'}) = 0 \\implies M' = M$"
          - td: "تحويل متطابق (كل نقطة هي صورة نفسها)."
        - - th: "$\\theta = \\pi$"
          - td: "$\\Omega M' = \\Omega M$ و $(\\vec{\\Omega M}, \\vec{\\Omega M'}) = \\pi \\implies \\vec{\\Omega M'} = -\\vec{\\Omega M}$"
          - td: "تناظر مركزي مركزه النقطة $\\Omega$."
        - - th: "$\\theta = \\frac{\\pi}{2}$"
          - td: "المثلث $\\Omega MM'$ قائم في $\\Omega$ ومتساوي الساقين في الاتجاه المباشر."
          - td: "دوران ربع دورة في الاتجاه الموجب."
        - - th: "$\\theta = -\\frac{\\pi}{2}$"
          - td: "المثلث $\\Omega MM'$ قائم في $\\Omega$ ومتساوي الساقين في الاتجاه غير المباشر."
          - td: "دوران ربع دورة في الاتجاه السالب."

  - h3: "3. خطوات الإنشاء الهندسي لصورة نقطة بالدوران"
    id: "rotation-construction"

  - parags:
      - rtl: "لإنشاء صورة النقطة $M$ بالدوران $r(\\Omega, \\theta)$، نتبع الخطوات التالية:"

  - tableLines:
      dir: rtl
      lines:
        - - th: "الخطوة"
          - th: "طريقة العمل والإنشاء"
          - th: "التعليل الرياضي"
        - - th: "1. رسم قوس الدوران"
          - td: "نضع شوكة الفرجار في المركز $\\Omega$ وقلم الرسم في النقطة $M$ ونرسم قوساً من دائرة."
          - td: "الشرط $\\Omega M' = \\Omega M$ يفرض وقوع النقطة الصورة على الدائرة التي مركزها $\\Omega$ ونصف قطرها $\\Omega M$."
        - - th: "2. تعيين اتجاه وقيس الزاوية"
          - td: "باستعمال المنقلة، نقيس الزاوية $\\theta$ انطلاقاً من نصف المستقيم $[\\Omega M)$ في الاتجاه المباشر (إذا كانت $\\theta > 0$) أو غير المباشر (إذا كانت $\\theta < 0$)."
          - td: "إشارة وقيمة الزاوية تحددان الموضع الدقيق لنصف المستقيم $[\\Omega M')$."
        - - th: "3. تحديد النقطة الصورة"
          - td: "نقطة تقاطع ضلع الزاوية الجديد مع القوس المرسوم هي النقطة $M'$."
          - td: "تقاطع الدائرة مع نصف المستقيم يحدد نقطة وحيدة تحقق شرطي المسافة والزاوية."

  - h3: "4. مثال تطبيقي (إنشاء صورة نقطة بالدوران)"
    id: "rotation-example"

  - parags:
      - rtl: "لتكن $\\Omega$ و $A$ نقطتين من المستوي حيث $\\Omega A = 3\\text{ cm}$. أنشئ النقطة $A'$ صورة $A$ بالدوران $r$ الذي مركزه $\\Omega$ وزاويته $\\theta = \\frac{\\pi}{3}$ ($60^\\circ$)."

  - tableLines:
      dir: rtl
      lines:
        - - th: "المرحلة"
          - th: "خطوات الإنشاء والحساب"
          - th: "النتيجة الهندسية"
        - - th: "1. إنشاء المسار الدائري"
          - td: "نرسم قوساً دائرياً مركزه $\\Omega$ ويمر بالنقطة $A$ في الاتجاه المباشر."
          - td: "تحقيق شرط ثبات المسافة $\\Omega A' = \\Omega A = 3\\text{ cm}$."
        - - th: "2. تعيين الزاوية"
          - td: "نضع مركز المنقلة على $\\Omega$ وخط الصفر على $[\\Omega A)$ ونعلم زاوية قيسها $60^\\circ$ في عكس اتجاه عقارب الساعة."
          - td: "تحقيق شرط الزاوية الموجهة $(\\vec{\\Omega A}, \\vec{\\Omega A'}) = \\frac{\\pi}{3}$."
        - - th: "3. تحديد طبيعة المثلث الناتجة"
          - td: "نصل بين $\\Omega$ و $A$ و $A'$. بما أن $\\Omega A = \\Omega A'$ والزاوية بينهما $60^\\circ$، فإن المثلث $\\Omega AA'$ متساوي الأضلاع."
          - td: "$A'$ هي الصورة، والمثلث $\\Omega AA'$ متساوي الأضلاع طول ضلعه $3\\text{ cm}$."
