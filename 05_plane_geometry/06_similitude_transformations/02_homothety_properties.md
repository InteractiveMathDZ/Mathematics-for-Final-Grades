elements:
  - h3: "1. الأثر الشعاعي للتحاكي (حفظ الاستقامية والخطية)"
    id: "homothety-vector-properties"

  - parags:
      - rtl: "التحاكي $h$ الذي مركزه $\\Omega$ ونسبته $k$ هو تحويل نقطي يحافظ على العلاقات الشعاعية والاستقامية. إذا كانت $A'$ و $B'$ هما الصورتان النقطيتان للنقطتين $A$ و $B$ على الترتيب بواسطة التحاكي $h$، فإن:"
      - center: "$$\\vec{A'B'} = k \\cdot \\vec{AB}$$"
      - rtl: "<strong>النتيجة الهندسية:</strong> ينتج عن هذه العلاقة أن المستقيم $(A'B')$ يوازي دائماً المستقيم $(AB)$."

  - h3: "2. أثر نسبة التحاكي على الأطوال، المساحات، والحجوم"
    id: "homothety-scaling-effects"

  - parags:
      - rtl: "يؤدي التحاكي $h(\\Omega, k)$ إلى تعديل مقاييس الأشكال الهندسية تبعاً للقيمة المطلقة لنسبته $|k|$ وفق القواعد الرياضية التالية:"

  - tableLines:
      dir: rtl
      lines:
        - - th: "المقدار الهندسي"
          - th: "علاقة التحويل بالتحاكي"
          - th: "معامل التغير"
        - - th: "الأطوال والمسافات"
          - td: "$A'B' = |k| \\times AB$"
          - td: "تُضرب الأطوال في $|k|$"
        - - th: "المساحات"
          - td: "$\\text{Area}' = k^2 \\times \\text{Area}$"
          - td: "تُضرب المساحات في $k^2$"
        - - th: "الحجوم"
          - td: "$\\text{Volume}' = |k|^3 \\times \\text{Volume}$"
          - td: "تُضرب الحجوم في $|k|^3$"

  - h3: "3. الخواص الإيجابية والسلبية لحفظ الأشكال"
    id: "homothety-preservation"

  - tableLines:
      dir: rtl

          - th: "المقادير غير المحفوظة (تتغير)"

          - td: "* الأطوال والمسافات بين النقط (إلا إذا كان $|k|=1$).<br>* مساحات الأشكال المستوية.<br>* حجوم المجسمات الهندسة."

  - h3: "4. مثال تطبيقي (حساب مساحة مثلث ناتج عن تحاكٍ)"
    id: "homothety-prop-example"

  - parags:

        - - th: "2. التعويض العددي"
