elements:
  - h3: "1. مفهوم الارتباط الخطي"
    id: "lin-dep-concept"

  - parags:
      - rtl: "نقول عن شعاعين غير معدومين $\\vec{u}$ و $\\vec{v}$ أنهما مرتبطان خطياً إذا كان لهما نفس المنحى (أي أن حاملاهما متوازيان أو منطبقان). رياضيا: يوجد عدد حقيقي $k$ يحقق العلاقة التالية:"
      - center: "$$\\vec{v} = k \\cdot \\vec{u}$$"

  - h3: "2. المحدّد:"
    id: "determinant-rule"

  - parags:
      - rtl: "لتفادي البحث العشوائي عن العدد $k$، نستخدم أداة 'المحدد' (Determinant) لمركبات الشعاعين $\\vec{u}=\\binom{X}{Y}$ و $\\vec{v}=\\binom{X'}{Y'}$. يحسب كالتالي:"
      - center: "$$\\det(\\vec{u}, \\vec{v}) = \\begin{vmatrix} X & X' \\\\ Y & Y' \\end{vmatrix} = X \\cdot Y' - Y \\cdot X'$$"

  - tableLines:
      dir: rtl
      lines:
        - - th: "قيمة المحدد"
          - th: "الوضعية الجبرية"
          - th: "الحكم الهندسي"
        - - th: "$\\det(\\vec{u}, \\vec{v}) = 0$"
          - td: "الشعاعان مرتبطان خطياً."
          - td: "الحوامل متوازية تماماً أو منطبقة."
        - - th: "$\\det(\\vec{u}, \\vec{v}) \\neq 0$"
          - td: "الشعاعان غير مرتبطين خطياً."
          - td: "الحوامل متقاطعة في نقطة فريدة حتماً."

  - h3: "3. تطبيقات الارتباط: التوازي والاستقامية"
    id: "lin-dep-apps"

  - parags:
      - rtl: "الارتباط الخطي هو الوسيلة الحسابية الأقوى لإثبات نظريات الوضع النسبي بدون الحاجة لأدوات القياس البصرية:"
      - rtl: "- <strong>إثبات توازي مستقيمين:</strong> يكون $(AB) // (CD)$ إذا وفقط إذا كان الشعاعان $\\vec{AB}$ و $\\vec{CD}$ مرتبطين خطياً ($\\det=0$)."
      - rtl: "- <strong>إثبات استقامية نقط ثلاث:</strong> تكون النقط $A, B, C$ في استقامية إذا وفقط إذا كان الشعاعان $\\vec{AB}$ و $\\vec{AC}$ مرتبطين خطياً ($\\det=0$)، لوجود الرأس المشترك $A$."

  - h3: "4. مثال تطبيقي (إثبات شرط الاستقامية)"
    id: "lin-dep-ex"

  - parags:
      - rtl: "لتكن النقط $A(1, 1)$، $B(3, 5)$، و $C(5, y_C)$. أوجد قيمة الترتيبة $y_C$ لكي تكون النقط $A, B, C$ في استقامية تامة."

  - tableLines:
      dir: rtl
      lines:
        - - th: "المرحلة"
          - th: "الحساب وتوظيف المحدد"
          - th: "النتيجة"
        - - th: "1. صياغة الأشعة"
          - td: "$\\vec{AB} = \\binom{3-1}{5-1} = \\binom{2}{4}$<br>$\\vec{AC} = \\binom{5-1}{y_C-1} = \\binom{4}{y_C-1}$"
          - td: "تجهيز المركبات"
        - - th: "2. تطبيق شرط الارتباط"
          - td: "$\\det(\\vec{AB}, \\vec{AC}) = 0 \\implies 2(y_C - 1) - 4(4) = 0$"
          - td: "معادلة بمجهول واحد"
        - - th: "3. عزل المجهول"
          - td: "$2y_C - 2 - 16 = 0 \\implies 2y_C = 18 \\implies y_C = 9$"
          - td: "$y_C = 9$"
