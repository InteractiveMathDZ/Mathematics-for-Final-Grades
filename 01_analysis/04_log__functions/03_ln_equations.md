elements:
  - h3: "1. القاعدة الأساسية للحل"
    id: "fundamental-solving-rule"

  - parags:
      - rtl: "تعتمد حلول المعادلات اللوغاريتمية على خاصية 'وحدانية' الدالة؛ أي أنها دالة متزايدة تماماً."
      - center: "$\\ln(a) = \\ln(b) \\iff a = b$"
      - center: "$\\ln(a) > \\ln(b) \\iff a > b$"
      - rtl: "بشرط أن يكون $a > 0$ و $b > 0$."

  - h3: "2. بروتوكول الحل (خطوات لا يمكن تجاوزها)"
    id: "solving-protocol"

  - tableLines:
      dir: rtl
      lines:
        - - th: "المرحلة"
          - th: "الإجراء المطلوب"
        - - td: "1. حيز الأمان"
          - td: "تحديد مجموعة التعريف $D$ (ما داخل اللوغاريتم موجباً تماماً)."
        - - td: "2. التجميع"
          - td: "استخدام الخصائص الجبرية لجعل المعادلة من الشكل $\\ln(u) = \\ln(v)$."
        - - td: "3. الحذف"
          - td: "نزع اللوغاريتم وحل المعادلة الناتجة $u = v$."
        - - td: "4. الفحص"
          - td: "مقارنة الحلول المستخرجة مع حيز الأمان $D$ (قبول أو رفض)."

  - h3: "3. مثال تطبيقي (فخ القيمة المرفوضة)"
    id: "example-solving"

  - parags:
      - rtl: "حل في $\\mathbb{R}$ المعادلة:"
      - center: "$\\ln(x) + \\ln(x-1) = \\ln(2)$"

  - tableLines:
      dir: rtl
      lines:
        - - th: "مجموعة التعريف"
          - td: "$x>0$ و $x-1>0 \\implies D = ]1; +\\infty[$"
        - - th: "التجميع"
          - td: "$\\ln(x(x-1)) = \\ln(2) \\implies x^2 - x = 2$"
        - - th: "الحل الجبري"
          - td: "$x^2 - x - 2 = 0 \\implies (x-2)(x+1) = 0$"
        - - th: "النتيجة"
          - td: "$x=2$ (مقبول) ، $x=-1$ (مرفوض لأنه خارج $D$)"

  - NB:
      title: "تنبيه:"
      type: "danger"
      NBs:
        - rtl: "أكبر خطأ يقع فيه الطالب هو البدء بالحل قبل تحديد مجموعة التعريف. تذكر أن اللوغاريتم لا يقبل النتائج السالبة حتى لو كانت حسابياً صحيحة!"

  - h3: "4. المتراجحات وتغيير الاتجاه"
    id: "ln-inequalities"

  - parags:
      - rtl: "عند حل المتراجحات، تذكر أن:"
      - center: "$\\ln(x) > 0$"
      - rtl: "يعني:"
      - center: "$x > 1$"
      - rtl: "وأن:"
      - center: "$\\ln(x) < 0$"
      - rtl: "يعني:"
      - center: "$0 < x < 1$."

  - NB:
      title: "دخول الأساس في الحل:"
      NBs:
        - rtl: "إذا واجهت معادلة من الشكل:"
        - center: "$\\ln(x) = a$،"
        - rtl: "فإن الحل هو:"
        - center: "$x = e^a$"
        - rtl: "(باستخدام الدالة العكسية التي سندرسها لاحقاً)."
