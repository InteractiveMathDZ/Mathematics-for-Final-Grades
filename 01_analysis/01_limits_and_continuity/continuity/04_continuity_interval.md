  - h3: "1. تعريف الاستمرارية على مجال مفتوح"
    id: "open-interval"

  - parags:
      - rtl: "نقول عن دالة $f$ أنها مستمرة على مجال مفتوح $]a, b[$ إذا كانت مستمرة عند كل نقطة من هذا المجال."
      - rtl: "هذا يعني أن المنحنى لا ينقطع في أي لحظة خلال هذه الفترة."

  - h3: "2. الاستمرارية على مجال مغلق $[a, b]$"
    id: "closed-interval"

  - parags:
      - rtl: "لكي تكون الدالة مستمرة على مجال مغلق، يجب أن يتحقق شرطان إضافيان عند الأطراف لضمان 'إغلاق' الطريق:"

  - tableLines:
      - - "الشرط"
        - "المعنى الرياضي"
      - - "الاستمرارية داخل المجال"
        - "مستمرة على المجال المفتوح $]a, b[$"
      - - "الاستمرارية عند الطرف $a$"
        - "$\\lim_{x \\to a^+} f(x) = f(a)$ (مستمرة يميناً)"
      - - "الاستمرارية عند الطرف $b$"
        - "$\\lim_{x \\to b^-} f(x) = f(b)$ (مستمرة يساراً)"

  - h3: "3. الربط بين الاستمرارية والاشتقاق (ملاحظة تمهيدية)"
    id: "cont-diff"

  - parags:
      - rtl: "كل دالة قابلة للاشتقاق على مجال هي بالضرورة مستمرة على هذا المجال. (الاشتقاق صفة أقوى من الاستمرارية)."
      - rtl: "لكن العكس غير صحيح دائماً؛ فقد تكون الدالة مستمرة ولكنها غير قابلة للاشتقاق (مثل دالة القيمة المطلقة عند الصفر)."

  - h3: "4. جدول تصنيف استمرارية المجالات"
    id: "interval-types"

  - tableLines:
      - - "نوع المجال"
        - "شرط الاستمرارية"
      - - "$[a, b]$"
        - "على $]a, b[$ + يميناً عند $a$ + يساراً عند $b$"
      - - "$[a, b[$"
        - "على $]a, b[$ + يميناً عند $a$"
      - - "$]a, b]$"
        - "على $]a, b[$ + يساراً عند $b$"
